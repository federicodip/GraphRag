    36      if ($fullPath.ToLower().Contains($root.ToLower())) { return $true }
    37    }
    38    return $false
    39  }

    41  function RelPath([string]$fullPath) {
    42    $root = (Get-Location).Path
    43    if ($fullPath.StartsWith($root)) { return ".\" + $fullPath.Substring($root.Length + 1) }

    47  function Wrap-Line([string]$line, [int]$width) {
    48    if ([string]::IsNullOrEmpty($line)) { return @('') }
    49    # soft wrap on whitespace up to $width
    50    return [regex]::Matches($line, "(.{1,$width})(\s+|$)") | ForEach-Object { $_.Groups[1].Value }
    51  }

    53  function Dump-File([IO.FileInfo]$f) {
    54    $rel = RelPath $f.FullName
    55    if ($f.Length -gt $MaxBytes) {
    56      Add-Line ""
    57      Add-Line "----- SKIP (too large) $rel ($($f.Length) bytes) -----"
    58      return
    59    }
    60  
    61    Add-Line ""
    62    Add-Line "----- BEGIN $rel -----"
    63  
    64    $text =
    65      if ($f.Extension -in @('.html','.htm')) {
    66        # strip giant data URIs; keep text readable
    67        (Get-Content -Raw -Encoding UTF8 $f.FullName) `
    68          -replace 'data:(?:image|font|application)/[a-zA-Z0-9.\+\-]+;base64,[A-Za-z0-9+/=\s]+' ,'[[data-uri-stripped]]'
    69      } else {
    70        Get-Content -Raw -Encoding UTF8 $f.FullName
    71      }
    72  
    73    $i = 1
    74    foreach ($ln in ($text -split "`r?`n")) {
    75      foreach ($w in (Wrap-Line $ln $WrapWidth)) {
    76        Add-Line ("{0,6}  {1}" -f $i, $w)
    77      }
    78      $i++
    79    }
    80  
    81    Add-Line "----- END $rel -----"
    82  }

    85  Remove-Item -ErrorAction Ignore $OutFile
    86  
    87  # --- Directory overview ----------------------------------------------------
    88  Add-Line "=== Directory Structure Overview ==="
    89  # build a simple tree (directories first, then files), excluding junk
    90  $items = Get-ChildItem -Recurse -Force | Where-Object {
    91    -not (Is-ExcludedPath $_.FullName)
    92  }
    93  
    94  # group by directory depth for pretty printing
    95  $rootPath   = (Get-Location).Path
    96  $itemsDirs  = $items | Where-Object { $_.PSIsContainer } | Sort-Object FullName
    97  $itemsFiles = $items | Where-Object { -not $_.PSIsContainer } | Sort-Object FullName
    98  
    99  # print directories
   100  foreach ($d in $itemsDirs) {
   101    $rel = RelPath $d.FullName
   102    $depth = ($rel -split '[\\/]').Count - 1
   103    $indent = ('|   ' * ($depth - 1)) + (if ($depth -gt 0) {'|-- '} else {''})
   104    Add-Line ($indent + $rel)
   105  }
   106  # print files
   107  foreach ($f in $itemsFiles) {
   108    $rel = RelPath $f.FullName
   109    $depth = ($rel -split '[\\/]').Count - 1
   110    $indent = ('|   ' * ($depth - 1)) + (if ($depth -gt 0) {'|-- '} else {''})
   111    Add-Line ($indent + $rel)
   112  }
   113  
   114  # --- File contents ---------------------------------------------------------
   115  Add-Line ""
   116  Add-Line "=== File Contents (filtered, readable) ==="
   117  
   118  $filesToDump = Get-ChildItem -Recurse -File -Force |
   119    Where-Object {
   120      -not (Is-ExcludedPath $_.FullName) -and
   121      ($includeExt -contains $_.Extension.ToLower())
   122    } |
   123    Sort-Object FullName
   124

   127  }

    55      def _open_text():
    56          return gzip.open(path, "rt", encoding="utf-8") if path.suffix.lower() == ".gz" else path.open("rt", encoding="utf-8")

    58      # Peek a little to guess the container type

    76          # 2a) JSON-LD with @graph: [...places...]
    77          f = _open_text()
    78          yielded = False
    79          try:
    80              for place in ijson.items(f, "@graph.item"):
    81                  yielded = True
    82                  if isinstance(place, dict):
    83                      yield place
    84          finally:
    85              f.close()
    86          if yielded:

    90          f = _open_text()
    91          yielded = False
    92          try:
    93              for place in ijson.items(f, "places.item"):
    94                  yielded = True
    95                  if isinstance(place, dict):
    96                      yield place
    97          finally:

   100              return

   102          # 2c) GeoJSON FeatureCollection: {"features":[{"properties":{...}}, ...]}
   103          f = _open_text()
   104          yielded = False
   105          try:
   106              for feat in ijson.items(f, "features.item"):
   107                  props = (feat or {}).get("properties")
   108                  if isinstance(props, dict):

   110                      yield props
   111          finally:
   112              f.close()

   114              return

   116          # 2d) dict-of-id -> place
   117          f = _open_text()
   118          yielded = False
   119          try:
   120              for key, place in ijson.kvitems(f, ""):
   121                  if isinstance(place, dict):
   122                      yielded = True
   123                      yield place
   124          finally:
   125              f.close()
   126          if yielded:
   127              return

   129      # Case 3: NDJSON fallback (one JSON object per line)
   130      f = _open_text()

   138              except json.JSONDecodeError:

   140              if isinstance(obj, dict):
   141                  yield obj

   152      alt, langs = [], []

   157              v = n.get(k)

   159                  parts = [p.strip() for p in v.split(",") if p.strip()]

   165          v = place.get(k)
   166          if isinstance(v, str) and v.strip():
   167              alt.append(v.strip())
   168      seen = set()
   169      out = []
   170      for a in alt:
   171          if a not in seen:

   230                  if not to_pid:
   231                      continue
   232                  sess.run(CYPHER_UPSERT_STUB, pleiadesId=to_pid, uri=to_uri)
   233                  sess.run(CYPHER_CONNECT,
   234                           **{"from": pid, "to": to_pid},
   235                           connectionType=c.get("connectionType"),
   236                           title=c.get("title"),
   237                           associationCertainty=c.get("associationCertainty"),
   238                           uri=c.get("uri"))
   239                  n_edges += 1
   240  
   241      driver.close()
   242      print(f"Ingested places: {n_places}, connections: {n_edges}")

     1  #!/usr/bin/env python3
     2  # -*- coding: utf-8 -*-
     3  """
     4  Convert a JSON array of text chunks (isaw.txt) into JSONL,
     5  decoding bare 'uXXXX' / 'uXXXXXX' codepoint strings into real Unicode.
     6  
     7  Usage:
     8    python to_jsonl_fix_unicode.py isaw.txt isaw.jsonl --article-id isaw-papers-1-2011
     9  """
    10  
    11  import argparse
    12  import json
    13  import re
    14  import sys
    15  import unicodedata
    16  from pathlib import Path
    17  
    18  # matches: u2643, u00b0, u1f45, u03b1, etc. (4–6 hex digits)
    19  UHEX = re.compile(r"u([0-9a-fA-F]{4,6})")
    20  
    21  
    22  def decode_bare_u_sequences(s: str) -> str:
    23      r"""
    24      Robustly decode literal \uXXXX sequences in plain text.
    25      - First collapse UTF-16 surrogate pairs to a single codepoint.
    26      - Then decode single \uXXXX (exactly 4 hex digits).
    27      - Anything invalid/out-of-range is left untouched.
    28      """
    29      # 1) Surrogate pairs: \uD800–\uDBFF followed by \uDC00–\uDFFF
    30      def _pair_repl(m):
    31          hi = int(m.group(1), 16)
    32          lo = int(m.group(2), 16)
    33          cp = 0x10000 + ((hi - 0xD800) << 10) + (lo - 0xDC00)
    34          try:
    35              return chr(cp)
    36          except ValueError:
    37              return m.group(0)
    38  
    39      s = re.sub(
    40          r'\\u(d[89ab][0-9a-f]{2})\\u(d[cdef][0-9a-f]{2})',
    41          lambda m: _pair_repl(m),
    42          s,
    43          flags=re.IGNORECASE
    44      )
    45  
    46      # 2) Single \uXXXX (exactly 4 hex digits)
    47      def _single_repl(m):
    48          cp = int(m.group(1), 16)
    49          # skip lone surrogates; they were handled above as pairs
    50          if 0xD800 <= cp <= 0xDFFF or cp > 0x10FFFF:
    51              return m.group(0)
    52          try:
    53              return chr(cp)
    54          except ValueError:
    55              return m.group(0)
    56  
    57      s = re.sub(
    58          r'\\u([0-9a-f]{4})',
    59          lambda m: _single_repl(m),
    60          s,
    61          flags=re.IGNORECASE
    62      )
    63  
    64      return s
    65  
    66  
    67  def normalize_text(s: str) -> str:
    68      """Unicode NFC + replace NBSP with regular space."""
    69      s = s.replace("\u00A0", " ")
    70      return unicodedata.normalize("NFC", s)
    71  
    72  
    73  def clean_text(s: str) -> str:
    74      """Decode bare-u sequences then normalize."""
    75      return normalize_text(decode_bare_u_sequences(s))
    76  
    77  
    78  def load_chunks(input_path: Path) -> list[str]:
    79      """Read a JSON array of strings from file."""
    80      raw = input_path.read_text(encoding="utf-8-sig")
    81      try:
    82          data = json.loads(raw)
    83      except json.JSONDecodeError as e:
    84          msg = (
    85              f"Failed to parse {input_path} as JSON.\n"
    86              "Ensure the file is a valid JSON array of strings, e.g. [\"...\", \"...\"]\n"
    87              f"JSON error: {e}"
    88          )
    89          raise SystemExit(msg)
    90      if not isinstance(data, list) or not all(isinstance(x, str) for x in data):
    91          raise SystemExit("Input must be a JSON array of strings.")
    92      return data
    93  
    94  
    95  def write_jsonl(chunks: list[str], output_path: Path, article_id: str):
    96      """Write JSONL with cleaned text (no raw text kept)."""
    97      with output_path.open("w", encoding="utf-8") as f:
    98          for i, text_raw in enumerate(chunks):
    99              text_clean = clean_text(text_raw)
   100              obj = {
   101                  "articleId": article_id,
   102                  "chunkId": f"{article_id}:{i:04d}",
   103                  "seq": i,
   104                  "text": text_clean,
   105              }
   106              f.write(json.dumps(obj, ensure_ascii=False) + "\n")
   107  
   108  
   109  def main():
   110      parser = argparse.ArgumentParser()
   111      parser.add_argument("input", type=Path, help="Path to isaw.txt (JSON array of strings)")
   112      parser.add_argument("output", type=Path, nargs="?", help="Path to output JSONL (default: input stem + .jsonl)")
   113      parser.add_argument("--article-id", default="isaw-article-1", help="Stable id/slug for this article")
   114      args = parser.parse_args()
   115  
   116      input_path: Path = args.input
   117      output_path: Path = args.output or input_path.with_suffix(".jsonl")
   118  
   119      chunks = load_chunks(input_path)
   120      write_jsonl(chunks, output_path, args.article_id)
   121  
   122      # quick sanity check for residual 'uXXXX' patterns in cleaned text
   123      residual = sum(1 for s in chunks if UHEX.search(clean_text(s)))
   124      if residual:
   125          print(f"WARNING: {residual} chunks still contain 'uXXXX' patterns after cleaning.", file=sys.stderr)
   126  
   127      print(f"Wrote {len(chunks)} lines to {output_path}")
   128  
   129  
   130  if __name__ == "__main__":
   131      main()
